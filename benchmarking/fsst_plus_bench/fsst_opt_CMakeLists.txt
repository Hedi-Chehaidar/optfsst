cmake_minimum_required(VERSION 3.10)
project(fsst_opt)

find_package(Threads REQUIRED)

# OptFSST sources are C++20.
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

include(CheckCXXCompilerFlag)
check_cxx_compiler_flag("-march=native" COMPILER_SUPPORTS_MARCH_NATIVE)
if(COMPILER_SUPPORTS_MARCH_NATIVE)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -march=native")
endif()

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

if(NOT MSVC)
    set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3 -DNDEBUG")
endif()

if(CMAKE_BUILD_TYPE STREQUAL Release)
    set_source_files_properties(fsst_avx512.cpp PROPERTIES COMPILE_FLAGS -O1)
endif()

# Library target name stays `fsst` so the parent CMakeLists links unchanged.
# fsst_create / fsst_compress are routed to Optfsst_* by --wrap at executable
# link time (see benchmarking/fsst_plus_bench/fsst_opt_wrap.cpp).
add_library(fsst
    libfsst.cpp
    fsst_avx512.cpp
    fsst_avx512_unroll1.inc
    fsst_avx512_unroll2.inc
    fsst_avx512_unroll3.inc
    fsst_avx512_unroll4.inc)
