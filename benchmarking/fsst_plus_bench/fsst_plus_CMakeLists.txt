cmake_minimum_required(VERSION 2.8.12...3.29)
project(fsst-plus)

set(CMAKE_CXX_STANDARD 11)

include_directories(third_party/duckdb/src/include)
link_directories(third_party/duckdb/build/release/src)

# Variant switch: vanilla = upstream FSST, opt = OptFSST sources.
set(FSST_VARIANT "vanilla" CACHE STRING "Which FSST source to build: vanilla or opt")
if(FSST_VARIANT STREQUAL "opt")
    message(STATUS "FSST+ benchmark: using OptFSST sources (third_party/fsst_opt)")
    include_directories(third_party/fsst_opt)
    add_subdirectory(third_party/fsst_opt)
else()
    message(STATUS "FSST+ benchmark: using vanilla FSST sources (third_party/fsst)")
    include_directories(third_party/fsst)
    add_subdirectory(third_party/fsst)
endif()

include_directories(src/cleaving)
include_directories(src/util)
include_directories(src/block)

add_executable(bench_runner src/bench_runner.cpp)
target_link_libraries(bench_runner duckdb fsst)
set_target_properties(bench_runner PROPERTIES
    BUILD_RPATH "${CMAKE_SOURCE_DIR}/third_party/duckdb/build/release/src")

if(FSST_VARIANT STREQUAL "opt")
    target_sources(bench_runner PRIVATE src/fsst_opt_wrap.cpp)
    target_link_options(bench_runner PRIVATE
        "LINKER:--wrap=fsst_create"
        "LINKER:--wrap=fsst_compress")
endif()
